58 class CRYBinning {
59 
60 public:
61  //Nominal constructor
62  //Data format is expected to be
63  //
64  // binning myBinning = { 1.0 2.0 3.0 8.0}
65  //
66  // with monotonically increaing values describing
67  // the bin edges. The last value is the upper limit
68  // of the last bin, so there is one more entry required
69  // than the # of bins.
70  CRYBinning(std::string data="");
71 
72  //Destructor. CRYBinning owns _bins so delete it
73  ~CRYBinning() {delete _bins; _bins=0;}
74 
75  // Print to cout the name and optionally (according to printData)
76  // the binning as defined by _bins
77  void print(std::ostream& o, bool printData=false);
78 
79  //Returns the key
80  std::string name() {return _name;}
81 
82  //Direct access to the binning defintion
83  const std::vector<double>* bins() const {return _bins;}
84 
85  //Given x (value), determine the corresponding bin
86  // Return value is 0..N
87  // x is outside of the defined range, an assert will be thrown
88  int bin(double value);
89 
90  // Get the boundaries of this binning
91  double min() {return (*_bins)[0];}
92  double max() {return (*_bins)[_bins->size()-1];}
93 
94 private:
95  //Key for this binning structur
96  std::string _name;
97  //The bin defintions - For N bins, there are N+1 values
98  //corresponding to the lower and upper limits of each bin
99  //and assuming no gaps between bins
100  std::vector<double> *_bins;
101 
102  //Store the # of bins for conviencience only (_bins->size())
103  unsigned int _size;
104 };
